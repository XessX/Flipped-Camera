CameraDevice.GetDevices

public static CameraDevice[] GetDevices ();

Description

Get all camera devices. 

// Get a camera device
var cameraDevice = CameraDevice.GetDevices()[0];
// Check if it supports torch
if (cameraDevice.IsTorchSupported)
    // Put on the torch
    cameraDevice.TorchEnabled = true;
